    function HomeView() {
      const { posts, seenPosts, markSeen } = usePosts();
      const { user } = useAuth();
      const { navigate } = useRouter();
      const { openProfile, showToast } = useUI();
      const { settings } = useSettings();
      const { bgApp, bgCard, borderMain, shadowMd, textMain, textSub, textMuted } = useThemeStyles();
      const tick = useTick(60000); 
      
      const defaultUserTabs = ['자유', '질문', '공부', '급식'];
      const userBoards = user?.selectedBoards?.length > 0 ? user.selectedBoards : defaultUserTabs;
      
      const [activeTab, setActiveTab] = useState(() => localStorage.getItem('lastActiveTab') || '인기');

      useEffect(() => {
        localStorage.setItem('lastActiveTab', activeTab);
      }, [activeTab]);

      const sortedPosts = useMemo(() => {
        let sorted = [...posts].filter(p => {
          if (p.visibility === '우리 반' && (p.schoolId !== user?.school?.id || p.grade !== user?.grade || p.classNum !== user?.classNum)) return false;
          if (p.visibility === '우리 학교' && p.schoolId !== user?.school?.id) return false;
          if (p.visibility === '지역구') {
            const userRegion = user?.school?.address?.split(' ').slice(0, 2).join(' ');
            if (userRegion && p.region !== userRegion) return false;
          }
          
          if (activeTab && p.boards) {
            if (!p.boards.includes(activeTab)) {
              if (activeTab !== '인기' && activeTab !== 'NOW') return false; 
            }
          }
          return true;
        });

        if (activeTab === 'NOW') {
          sorted.sort((a, b) => (b.createdAt?.toMillis() || 0) - (a.createdAt?.toMillis() || 0));
        } else {
          sorted.sort((a, b) => {
            const aSeen = seenPosts.has(a.id); 
            const bSeen = seenPosts.has(b.id);
            if (aSeen === bSeen) {
              const aLikes = a.likedBy?.length || 0;
              const bLikes = b.likedBy?.length || 0;
              if (aLikes === bLikes) {
                return (b.createdAt?.toMillis() || 0) - (a.createdAt?.toMillis() || 0);
              }
              return bLikes - aLikes;
            }
            return aSeen ? 1 : -1;
          });
        }
        return sorted;
      }, [posts, seenPosts, activeTab, user]); 

      const handleVote = async (e, postId, optionIndex, currentPoll, hasVoted) => {
        e.stopPropagation();
        if (hasVoted) return showToast('이미 투표에 참여했습니다.');
        await voteOnPost(postId, optionIndex, currentPoll, user?.uid);
        const voted = JSON.parse(localStorage.getItem('votedPosts') || '[]');
        localStorage.setItem('votedPosts', JSON.stringify([...voted, postId]));
        showToast('투표가 반영되었습니다!');
      };

      return (
        <div className={`flex-1 flex flex-col animate-view relative h-full ${bgApp}`}>
          <div className={`p-5 flex justify-between items-center z-20 ${bgApp} border-b md:border-none border-gray-200`}>
            <h1 className={`sys-text font-black flex items-center gap-2 ${textMain}`}>
              <div className="animate-view md:hidden">
                <div className={`logo-fixed flex items-center justify-center rotate-3`}>
                  <span className="logo-text-fixed">OCL</span>
                </div>
              </div>
              홈 피드
            </h1>
          </div>
          
          {/* 구독 게시판들이 하나 이상인 경우 헤더를 해당 포맷으로 노출 */}
          <div className={`flex gap-2 px-5 pb-2 overflow-x-auto hide-scrollbar z-10 w-full pt-1 md:border-b border-gray-200 ${bgApp}`}>
            {['인기', 'NOW', ...userBoards].map(b => (
              <button 
                key={b} 
                onClick={() => setActiveTab(b)} 
                className={`px-5 py-2 rounded-full border-2 font-extrabold text-sm transition-all whitespace-nowrap flex-shrink-0 ${borderMain} ${activeTab === b ? (b === 'NOW' ? `bg-[#FFD900] text-black ${shadowMd}` : `main-btn ${shadowMd}`) : `bg-transparent ${textMain} hover:opacity-70`}`}
              >
                {b === '인기' ? '인기 🔥' : b}
              </button>
            ))}
            
            {/* [+] 게시판 구독 추가 버튼 */}
            <button onClick={() => navigate('search')} className={`px-4 py-2 rounded-full border-2 border-dashed font-black text-sm whitespace-nowrap flex-shrink-0 ${borderMain} hover:opacity-80`}>
              + 탐색
            </button>
          </div>
          
          <div className="flex-1 overflow-y-auto px-5 pt-4 flex flex-col gap-4 hide-scrollbar pb-6">
            {sortedPosts.length === 0 ? (
              <div className={`flex flex-col items-center justify-center py-20 ${textMuted}`}>
                 <Icon name="file-text" size={48} className="mb-4 opacity-50"/>
                 <p className="font-bold text-center">아직 게시글이 없어요.<br/>첫 글을 남겨보세요!</p>
              </div>
            ) : (
              sortedPosts.map(p => {
                const isLiked = user && p.likedBy?.includes(user.uid);
                const bestComments = p.comments?.slice().sort((a, b) => {
                  if (a.id === p.acceptedCommentId) return -1;
                  if (b.id === p.acceptedCommentId) return 1;
                  return (b.likes || 0) - (a.likes || 0);
                }).slice(0, Math.floor(Math.random() * 2) + 1);
                
                const hasVoted = p.poll && p.poll.options.some(opt => opt.voters?.includes(user?.uid));
                const myVoteIndex = p.poll?.options.findIndex(opt => opt.voters?.includes(user?.uid));

                return (
                  <div 
                    key={p.id} 
                    onDoubleClick={() => {
                      if(user) { 
                        togglePostLike(p.id, user.uid, isLiked); 
                        if(navigator.vibrate) navigator.vibrate(50); 
                      }
                    }}
                    onClick={() => { markSeen(p.id); navigate(`post?id=${p.id}`); }} 
                    className={`${bgCard} border-2 ${borderMain} p-4 ${shadowMd} cursor-pointer hover:-translate-y-1 hover:shadow-[6px_6px_0px_rgba(0,0,0,1)] transition-all select-none ${settings.feedStyle === 'mini' ? 'py-3' : 'rounded-2xl'}`}
                  >
                    <div className="flex justify-between items-start mb-2">
                      <div className="flex gap-1 flex-wrap">
                        {p.tags?.map(tag => (

                      <span className={`text-xs font-bold ${textMuted}`}>{formatTime(p.createdAt)}</span>
                    </div>
                    
                    {settings.feedStyle === 'thumbnail' && p.image && (
                       <div className={`w-full h-40 md:h-56 border-2 rounded-xl mb-3 overflow-hidden ${borderMain}`}>
                         <img src={p.image} alt="thumbnail" className="w-full h-full object-cover" />
                       </div>
                    )}
                    
                    <h3 className={`font-black text-lg truncate mb-1 ${textMain}`}>{p.title}</h3>
                    <p className={`text-sm font-bold ${textSub} ${settings.feedStyle === 'mini' ? 'line-clamp-2' : 'line-clamp-10'}`}>{p.content}</p>

                    {p.poll && settings.feedStyle !== 'mini' && (
                      <div className={`mt-3 mb-1 p-3 rounded-xl border flex flex-col gap-2 bg-gray-50 border-gray-200`}>
                         <div className={`text-xs font-bold flex items-center gap-1 mb-1 ${textSub}`}><Icon name="bar-chart-2" size={14}/> {p.poll.question}</div>
                         {p.poll.options.map((o, i) => {
                           const isMyVote = myVoteIndex === i;
                           return (
                             <div key={i} onClick={(e) => handleVote(e, p.id, i, p.poll, hasVoted)} className={`relative h-10 border rounded-lg flex items-center px-3 cursor-pointer overflow-hidden bg-white border-black hover:bg-gray-100`}>
                               <div className={`absolute left-0 top-0 bottom-0 transition-all duration-500 ease-out ${hasVoted ? (isMyVote ? 'main-bg opacity-100' : 'bg-gray-200 opacity-50') : 'main-bg opacity-20'}`} style={{ width: `${o.percent || 0}%` }}></div>
                               <span className={`relative z-10 font-black text-sm w-full flex justify-between items-center ${textMain} ${hasVoted && !isMyVote ? 'opacity-50' : ''}`}>
                                 <span>{o.label}</span>
                                 <span>{o.percent || 0}%</span>
                               </span>
                             </div>
                           );
                         })}
                      </div>
                    )}
                    
                    <div className="flex justify-between items-center mt-4">
                      <span 
                        onClick={(e) => { 
                          e.stopPropagation(); 
                          if(p.authorId && !p.isAnon) openProfile(p.authorId); 
                        }} 
                        className={`text-xs font-bold flex items-center gap-1 text-gray-800 ${p.authorId && !p.isAnon ? 'hover:underline cursor-pointer' : ''}`}
                      >
                        <div className={`w-6 h-6 border rounded-full flex items-center justify-center overflow-hidden ${borderMain} bg-gray-200`}>
                          {p.profileImage && !p.isAnon ? <img src={p.profileImage} alt="Profile" className="w-full h-full object-cover"/> : <Icon name="user" size={14} className={textMuted}/>}
                        </div>
                        <span className={textMain}>{p.author}</span>
                      </span>
                      <div className={`flex gap-4 text-sm font-bold ${textSub}`}>
                        <button onClick={(e) => { e.stopPropagation(); if(user) togglePostLike(p.id, user.uid, isLiked); }} className={`flex items-center gap-1.5 transition-transform ${isLiked ? 'text-[#FF4444] scale-110' : 'hover:scale-110'}`}>
                          <Icon name="heart" size={18} fill={isLiked ? "currentColor" : "none"} strokeWidth={3} /> 
                          <AnimatedCount value={p.likedBy?.length || 0} />
                        </button>
                        <span className="flex items-center gap-1.5 hover:text-black transition-colors">
                          <Icon name="message-square" size={16}/>
                          <AnimatedCount value={p.commentsCount || 0} />
                        </span>
                      </div>
                    </div>
                    
                    {bestComments?.length > 0 && settings.feedStyle !== 'mini' && (
                      <div className={`mt-4 border-t pt-3 pb-2 flex flex-col gap-1 -mx-4 px-4 -mb-4 rounded-b-2xl bg-gray-50 border-gray-200`}>
                        {bestComments.map((c, i) => (
                          <div key={i} className="flex gap-2 items-start py-0.5">
                            <span className={`text-xs font-bold line-clamp-1 break-all ${textMain}`}>
                              {c.id === p.acceptedCommentId && <span className="text-[#00E066] font-black mr-1">✓</span>}
                              <span className={`font-black mr-2 ${textMuted}`}>{c.author}</span>{c.text}
                            </span>
                          </div>
                        ))}
                        <div className="h-1"/>
                      </div>
                    )}
                  </div>
                );
              })
            )}
          </div>
        </div>
      );
    }

      const [expandedCategory, setExpandedCategory] = useState(null);

        getDocs(collection(db, 'users')).then(snap => {
          setAllUsers(snap.docs.map(d => ({ uid: d.id, ...d.data() })));
        });
      }, []);

      const k = keyword.trim().toLowerCase();

                </div>

                  const defaultItems = defaultCat ? defaultCat.items : [];
                  
                  const normalizedDefault = defaultItems.map(item => {
                    if (typeof item === 'string') return { name: item, subs: [], isDefault: true };
                    return { name: item.name, subs: item.subs || [], isDefault: true };

                  const customItems = catBoards.map(b => ({ name: b.name, desc: b.description, subs: [], image: b.image, isDefault: false }));

                  if (allItems.length === 0) return null;

                      <button

                                    <span className={`text-base font-black ${textMain}`}>{item.name}</span>
                                    <span className={`text-xs font-bold ${textMuted}`}>{item.desc || `${cat} 소속`}</span>

    // [ views/BoardView.jsx ] 개별 게시판 화면

    function BoardView() {
      const { params, navigate, goBack } = useRouter();
      const { posts, seenPosts, markSeen } = usePosts();
      const { user } = useAuth();
      const { showToast, openProfile } = useUI();
      const { settings } = useSettings();
      const { bgApp, bgCard, borderMain, shadowMd, textMain, textSub, textMuted } = useThemeStyles();
      const tick = useTick(60000); 

      const boardName = params.id ? decodeURIComponent(params.id) : '자유';
      const isGlobalHot = boardName === '인기';
      const [tab, setTab] = useState(isGlobalHot ? '인기' : '최신');
      
      const sportsTeams = useMemo(() => getSportsTeams(boardName), [boardName]);
      const [selectedTeamTab, setSelectedTeamTab] = useState('전체');

      const isSubscribed = user?.selectedBoards?.includes(boardName);

      const handleToggleSubscribe = async (e) => {
        e.stopPropagation();
        if (!user) return; 
        const currentBoards = user.selectedBoards || [];
        let newBoards;
        if (isSubscribed) {
          newBoards = currentBoards.filter(b => b !== boardName);
          showToast(`${boardName} 게시판 구독을 취소했습니다.`);
        } else {
          newBoards = [...currentBoards, boardName];
          showToast(`${boardName} 게시판을 구독했습니다!`);
        }
        await updateUserProfile(user.uid, { selectedBoards: newBoards });
      };

      const handleToggleTeamJoin = async (e, team) => {
        e.stopPropagation();
        if (!user) return; 
        const currentTeams = user.selectedTeams || [];
        const newTeams = currentTeams.includes(team) ? currentTeams.filter(t => t !== team) : [...currentTeams, team];
        await updateUserProfile(user.uid, { selectedTeams: newTeams });
        showToast(`'${team}' 팀 구독이 변경되었습니다.`);
      };

      const sortedPosts = useMemo(() => {
        let sorted = [...posts].filter(p => {
          if (p.visibility === '우리 반' && (p.schoolId !== user?.school?.id || p.grade !== user?.grade || p.classNum !== user?.classNum)) return false;
          if (p.visibility === '우리 학교' && p.schoolId !== user?.school?.id) return false;
          if (p.visibility === '지역구') {
            const userRegion = user?.school?.address?.split(' ').slice(0, 2).join(' ');
            if (userRegion && p.region !== userRegion) return false;
          }
          
          if (!isGlobalHot) {
            if (p.boards && Array.isArray(p.boards)) {
               if (!p.boards.includes(boardName)) return false;
            } else {
               if (p.boardId !== boardName && p.boardName !== boardName && !p.tags?.includes(boardName)) return false;
            }
          }

          if (sportsTeams.length > 0 && selectedTeamTab !== '전체') {
            if (!p.tags?.includes(selectedTeamTab)) return false;
          }

          return true;
        });

        if (tab === '최신') {
          sorted.sort((a, b) => (b.createdAt?.toMillis() || 0) - (a.createdAt?.toMillis() || 0));
        } else {
          sorted.sort((a, b) => {
            const aSeen = seenPosts.has(a.id); 
            const bSeen = seenPosts.has(b.id);
            if (aSeen === bSeen) {
              const aLikes = a.likedBy?.length || 0;
              const bLikes = b.likedBy?.length || 0;
              if (aLikes === bLikes) {
                return (b.createdAt?.toMillis() || 0) - (a.createdAt?.toMillis() || 0);
              }
              return bLikes - aLikes;
            }
            return aSeen ? 1 : -1;
          });
        }
        return sorted;
      }, [posts, seenPosts, tab, user, boardName, isGlobalHot, sportsTeams.length, selectedTeamTab]); 

      const handleVote = async (e, postId, optionIndex, currentPoll, hasVoted) => {
        e.stopPropagation();
        if (hasVoted) return showToast('이미 투표에 참여했습니다.');
        await voteOnPost(postId, optionIndex, currentPoll, user?.uid);
        showToast('투표가 반영되었습니다!');
      };

      return (
        <div className={`flex-1 flex flex-col animate-[slideInRight_0.3s_ease-out] relative h-full ${bgApp}`}>
          <div className={`p-4 md:p-6 border-b-2 flex items-center justify-between z-20 ${bgCard} ${borderMain} shadow-sm`}>
            <div className="flex items-center gap-3">
              <button onClick={() => goBack()} className={`p-2 -ml-2 rounded-full transition-colors ${textMain} hover:bg-gray-100 w-10 h-10 flex shrink-0 items-center justify-center`}>
                <Icon name="chevron-left" size={28} strokeWidth={3}/>
              </button>
              <span className={`sys-text font-black text-xl md:text-2xl ${textMain}`}>{boardName}</span>
            </div>
            {!isGlobalHot && (
              <button onClick={handleToggleSubscribe} className={`p-2 -mr-2 rounded-full transition-colors ${textMain} hover:bg-gray-100 w-10 h-10 flex shrink-0 items-center justify-center`}>
                {isSubscribed ? <Icon name="check" size={28} strokeWidth={3}/> : <Icon name="plus" size={28} strokeWidth={3}/>}
              </button>
            )}
          </div>

          {sportsTeams.length > 0 && (
            <div className={`flex gap-3 px-5 py-3 overflow-x-auto hide-scrollbar z-10 w-full border-b-2 ${borderMain} bg-gray-50`}>
              <button 
                 onClick={() => setSelectedTeamTab('전체')} 
                 className={`px-4 py-1.5 rounded-full border-2 text-sm font-black flex-shrink-0 transition-all ${borderMain} ${selectedTeamTab === '전체' ? `main-btn ${shadowMd}` : `bg-transparent ${textMain} opacity-50`}`}
              >
                 전체
              </button>
              {sportsTeams.map(t => {
                 const isJoined = user?.selectedTeams?.includes(t);
                 const isSelected = selectedTeamTab === t;
                 return (
                   <div key={t} className={`flex items-stretch rounded-full border-2 overflow-hidden flex-shrink-0 transition-all ${borderMain} ${isSelected ? shadowMd : ''}`}>
                     <button onClick={() => setSelectedTeamTab(t)} className={`px-4 py-1.5 text-sm font-black transition-colors ${isSelected ? 'bg-black text-white' : 'bg-white text-black'}`}>
                       {t}
                     </button>
                     <button onClick={(e) => handleToggleTeamJoin(e, t)} className={`px-3 py-1.5 text-sm font-black border-l-2 transition-colors flex items-center justify-center ${borderMain} ${isJoined ? 'bg-[#FFD900] text-black' : 'bg-gray-200 text-gray-500'}`}>
                       {isJoined ? <Icon name="check" size={16}/> : '가입'}
                     </button>
                   </div>
                 )
              })}
            </div>
          )}
          
          <div className={`flex gap-3 px-5 py-3 overflow-x-auto hide-scrollbar z-10 w-full border-b-2 ${borderMain}`}>
            <button onClick={() => setTab('인기')} className={`px-5 py-2 md:py-2.5 rounded-full border-2 font-extrabold text-sm md:text-base transition-all whitespace-nowrap flex-shrink-0 ${borderMain} ${tab === '인기' ? `main-btn ${shadowMd}` : `bg-transparent ${textMain} hover:opacity-70`}`}>
              인기 🔥
            </button>
            <button onClick={() => setTab('최신')} className={`px-5 py-2 md:py-2.5 rounded-full border-2 font-extrabold text-sm md:text-base transition-all ${borderMain} ${tab === '최신' ? `main-btn ${shadowMd}` : `bg-transparent ${textMain} hover:opacity-70`}`}>
              최신 ⚡

          </div>
          
          <div className="flex-1 overflow-y-auto px-5 pt-4 flex flex-col gap-4 hide-scrollbar pb-6 relative">
            {sortedPosts.length === 0 ? (
              <div className={`flex flex-col items-center justify-center py-20 ${textMuted}`}>

                 <p className="font-bold text-center text-lg">아직 게시글이 없어요.<br/>첫 글을 남겨보세요!</p>

                return (

    // [ views/SchoolView.jsx ] 우리 학교 정보 뷰 ("급식 랭킹"으로 이모티콘 제거 및 기능 강화)
    // =====================================================================
    function SchoolView() {
      const { user } = useAuth(); 
      const { showToast } = useUI();
      const { bgApp, bgCard, borderMain, shadowMd, shadowSm, textMain, textMuted, textSub } = useThemeStyles();
      
      const [activeTab, setActiveTab] = useState('info'); // 'info' | 'ranking'
      const [rankScope, setRankScope] = useState('national'); // 'national' | 'regional'
      
      const [meals, setMeals] = useState({}); 
      const [tt, setTt] = useState([]);
      const [todayYmd, setTodayYmd] = useState('');
      
      const [mealDocs, setMealDocs] = useState({}); // 내 학교 오늘 급식 문서들
      const [rankings, setRankings] = useState([]); // 오늘 전국/지역 급식 문서들

      const [isGenerating, setIsGenerating] = useState(false);

      // 날짜 및 API 데이터 호출
      useEffect(() => {
        if(!user?.school) return;
        const fetchD = async () => {
          const d = new Date(); 
          const ymd = d.toISOString().split('T')[0].replace(/-/g,'');
          setTodayYmd(ymd);
          
          if([0,6].includes(d.getDay())) return setMeals({ 'OFF': true });
          
          let parsedMeals = {};
          try { 
            const r = await fetch(`https://open.neis.go.kr/hub/mealServiceDietInfo?KEY=955297527f064afeb339cd55c97b0fcb&Type=json&ATPT_OFCDC_SC_CODE=${user.school.eduCode}&SD_SCHUL_CODE=${user.school.id}&MLSV_YMD=${ymd}`); 
            const data = await r.json(); 
            
            if (data.mealServiceDietInfo && data.mealServiceDietInfo[1].row) {
              data.mealServiceDietInfo[1].row.forEach(row => {
                const cleanMeals = row.DDISH_NM.split('<br/>').map(s => s.replace(/\\([^)]*\\)/g, '').replace(/[^가-힣\\s]/g, '').trim()).filter(Boolean);
                parsedMeals[row.MMEAL_SC_NM] = cleanMeals;
              });
              setMeals(parsedMeals);
              
              // Firestore 연동: 불러온 급식 데이터가 DB에 없으면 자동 생성
              for (const [mType, mList] of Object.entries(parsedMeals)) {
                const docId = `${user.school.id}_${ymd}_${mType}`;
                const ref = doc(getPublicCol('meal_votes'), docId);
                const snap = await getDoc(ref);
                if (!snap.exists()) {
                  await setDoc(ref, {
                    schoolId: user.school.id,
                    schoolName: user.school.name,
                    region: user.school.address.split(' ').slice(0,2).join(' '),
                    date: ymd,
                    mealType: mType,
                    menu: mList,
                    likes: [],
                    dislikes: [],
                    score: 0
                  });
                }
              }
            } else {
              setMeals({});
            }
          } catch(e) { console.error(e); }
          
          try { 
            let tT = 'hisTimetable';
            if (user.school.type.includes('중학교')) tT = 'misTimetable';
            if (user.school.type.includes('초등학교')) tT = 'elsTimetable';

            const r = await fetch(`https://open.neis.go.kr/hub/${tT}?KEY=955297527f064afeb339cd55c97b0fcb&Type=json&ATPT_OFCDC_SC_CODE=${user.school.eduCode}&SD_SCHUL_CODE=${user.school.id}&ALL_TI_YMD=${ymd}&GRADE=${user.grade}&CLASS_NM=${user.classNum}`); 
            const data = await r.json(); 
            
            if (data && data[tT] && data[tT][1] && data[tT][1].row) {
              // 교시 순서대로 정렬 (PERIO 정렬하여 시간표 정렬 문제 완전히 해결)
              const sorted = data[tT][1].row.sort((a,b) => Number(a.PERIO) - Number(b.PERIO));
              setTt(sorted.map(x=>x.ITRT_CNTNT)); 
            } else {
              setTt(['오늘 시간표 데이터가 없습니다.']);
            }
          } catch(e) { 
            setTt(['시간표를 불러오지 못했습니다.']); 
          }
        }; 
        fetchD();
      }, [user]);

      // 급식 랭킹 데이터 불러오기 (Firebase)
      useEffect(() => {
         if (!todayYmd) return;
         const unsub = onSnapshot(doc(getPublicCol('meal_rankings'), todayYmd), (snap) => {
             if(snap.exists()) {
                 setRankings(snap.data().data);
             } else {
                 setRankings(null);
             }
         });
         
         // 내 학교 급식 평가 현황 가져오기
         const q = query(getPublicCol('meal_votes'), where('date', '==', todayYmd), where('schoolId', '==', user?.school?.id || ''));
         const unsub2 = onSnapshot(q, (snap) => {
             const docs = snap.docs.map(d => ({ id: d.id, ...d.data() }));
             const myMap = {};
             docs.forEach(d => myMap[d.mealType] = d);
             setMealDocs(myMap);
         });

         return () => { unsub(); unsub2(); };
      }, [todayYmd, user]);

      const handleMealVote = async (mealType, voteType) => {
         if (!user) return;
         const docId = `${user.school.id}_${todayYmd}_${mealType}`;
         const ref = doc(getPublicCol('meal_votes'), docId);
         const snap = await getDoc(ref);
         if(!snap.exists()) return showToast('급식 정보를 서버에서 불러오는 중입니다. 잠시 후 시도해주세요.');

         const data = snap.data();
         let likes = data.likes || [];
         let dislikes = data.dislikes || [];

         const hasLiked = likes.includes(user.uid);
         const hasDisliked = dislikes.includes(user.uid);

         likes = likes.filter(uid => uid !== user.uid);
         dislikes = dislikes.filter(uid => uid !== user.uid);

         if (voteType === 'like' && !hasLiked) likes.push(user.uid);
         if (voteType === 'dislike' && !hasDisliked) dislikes.push(user.uid);

         await updateDoc(ref, {
             likes, dislikes, score: likes.length - dislikes.length
         });
      };

      const handleGenerateAIRanking = async () => {
         if (isGenerating) return;
         setIsGenerating(true);
         showToast('전국 급식 데이터를 수집 중입니다... (최대 10초 소요)');

         // 1. Fetch meals from multiple regions
         const regions = [
           {code:'B10', name:'서울'}, {code:'J10', name:'경기'}, 
           {code:'C10', name:'부산'}, {code:'D10', name:'대구'}, {code:'E10', name:'인천'}
         ];
         let mealDataList = [];
         
         for (const r of regions) {
            try {
              const res = await fetch(`https://open.neis.go.kr/hub/mealServiceDietInfo?KEY=955297527f064afeb339cd55c97b0fcb&Type=json&ATPT_OFCDC_SC_CODE=${r.code}&MLSV_YMD=${todayYmd}&pSize=4`);
              const data = await res.json();
              if (data.mealServiceDietInfo) {
                data.mealServiceDietInfo[1].row.forEach(row => {
                   const cleanMenu = row.DDISH_NM.replace(/<br\/>/g, ', ').replace(/[^가-힣\s,]/g, '').trim();
                   mealDataList.push(`[${r.name}] ${row.SCHUL_NM} (${row.MMEAL_SC_NM}): ${cleanMenu}`);
                });
              }
            } catch(e) { console.error("NEIS fetch error", e); }
         }

         if (mealDataList.length === 0) {
            showToast('오늘 전국에 조회된 급식 데이터가 없습니다 (주말/휴무일 등)');
            setIsGenerating(false);
            return;
         }

         showToast('AI가 급식 랭킹을 매기는 중입니다...');
         const prompt = `다음은 오늘 전국의 일부 고등학교 급식 메뉴야:\n${mealDataList.join('\n')}\n\n이 메뉴들을 평가해서 전체 전국 1~3위, 그리고 각 지역별 1위를 선정해줘. 평가 기준은 학생들의 선호도야.`;

         // 안전한 환경 변수 처리 구조 적용 (노출 방지)
         const apiKey = (window.process?.env?.REACT_APP_GEMINI_API_KEY) || (window.REACT_APP_GEMINI_API_KEY) || "";
         if (!apiKey) {
            showToast('Gemini API Key가 누락되었습니다. 랭킹 생성을 위해 API 키 설정을 확인해주세요.');
            setIsGenerating(false);
            return;
         }

         const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
         const payload = {
           systemInstruction: { parts: [{ text: "You are a school meal expert. Always output valid JSON." }] },
           contents: [{ parts: [{ text: prompt }] }],
           generationConfig: {
             responseMimeType: "application/json",
             responseSchema: {
               type: "OBJECT",
               properties: {
                 national: { type: "ARRAY", items: { type: "OBJECT", properties: { rank: { type: "INTEGER" }, schoolName: { type: "STRING" }, region: { type: "STRING" }, meal: { type: "STRING" } } } },
                 regional: { type: "ARRAY", items: { type: "OBJECT", properties: { region: { type: "STRING" }, schoolName: { type: "STRING" }, meal: { type: "STRING" } } } }
               }
             }
           }
         };

         try {
             let resultJson = null;
             // 지수 백오프 방식 재시도 로직
             for (let i = 0; i < 5; i++) {
                 try {
                     const res = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) });
                     if (!res.ok) throw new Error();
                     const data = await res.json();
                     resultJson = JSON.parse(data.candidates[0].content.parts[0].text);
                     break;
                 } catch (e) {
                     await new Promise(resolve => setTimeout(resolve, Math.pow(2, i) * 1000));
                 }
             }
             if (!resultJson) throw new Error("Gemini generation failed");

             // 2. Save Ranking to Firebase
             // 당일만 유지 후 자동 삭제 로직을 위해 매일 YMD 고유 ID로 갱신 저장
             await setDoc(doc(getPublicCol('meal_rankings'), todayYmd), { data: resultJson, createdAt: serverTimestamp() });

             // 3. Create a Post in Freedom/Meal Board (작성자: 급식 순위, 프로필사진: 🍱)
             let postContent = `오늘 AI가 선정한 전국 급식 랭킹입니다! 여러분의 생각은 어떠신가요? 🍱\n\n`;
             resultJson.national.forEach(n => {
                postContent += `🏆 ${n.rank}위: ${n.schoolName} (${n.region})\n메뉴: ${n.meal}\n\n`;
             });
             postContent += `--- 각 지역별 1위 ---\n`;
             resultJson.regional.forEach(r => {
                postContent += `📍 [${r.region}] ${r.schoolName}\n메뉴: ${r.meal}\n\n`;
             });

             await createNewPost({
                title: `[AI 선정] ${todayYmd.substring(4,6)}월 ${todayYmd.substring(6,8)}일 전국 급식 랭킹 🏆`,
                content: postContent,
                tags: ['급식 랭킹', '인기'],
                boards: ['급식', '자유', '메인'],
                boardId: '급식',
                boardName: '급식',
                author: '급식 순위',
                authorId: 'ai-bot-01',
                profileImage: 'https://cdn-icons-png.flaticon.com/512/3413/3413535.png', // 급식 이미지 대용
                visibility: '전국',
                isAnon: false,
                schoolId: null,
                region: '전국'
             });

             showToast('랭킹 분석 및 게시물 등록 완료!');
         } catch(e) {
             showToast('AI 분석 중 오류가 발생했습니다.');
             console.error(e);
         } finally {
             setIsGenerating(false);
         }
      };

      const isOff = meals['OFF'];
      const noMeals = Object.keys(meals).length === 0;

      const filteredRankings = rankings ? (rankScope === 'national' ? rankings.national : rankings.regional) : [];

      return (
        <div className={`flex-1 flex flex-col h-full animate-view relative ${bgApp}`}>
          <div className={`pt-6 px-6 pb-4 border-b-2 ${borderMain} shadow-sm z-10 ${bgCard}`}>
            <h1 className={`sys-text font-black mb-2 leading-tight ${textMain}`}>{user?.school?.name || '우리 학교'}</h1>
            <p className={`font-bold text-sm flex items-center gap-1 ${textMuted}`}><Icon name="graduation-cap" size={18}/> {user?.grade}학년 {user?.classNum}반 생활 정보</p>
            
            <div className="flex gap-2 mt-5">
               <button onClick={() => setActiveTab('info')} className={`flex-1 py-2 font-black text-sm rounded-xl border-2 transition-all ${borderMain} ${activeTab === 'info' ? `main-btn ${shadowMd}` : 'bg-transparent opacity-50'}`}>우리 학교 정보</button>
               <button onClick={() => setActiveTab('ranking')} className={`flex-1 py-2 font-black text-sm rounded-xl border-2 transition-all flex justify-center items-center gap-1 ${borderMain} ${activeTab === 'ranking' ? `bg-[#FFD900] text-black ${shadowMd}` : 'bg-transparent opacity-50'}`}>👑 급식 랭킹</button>
            </div>
          </div>

          <div className="flex-1 overflow-y-auto p-5 md:p-8 hide-scrollbar flex flex-col gap-6 pb-6">
            
            {activeTab === 'info' && (
              <>
                {/* 우리학교 급식 카드 영역 */}
                <div>
                  <h3 className={`font-black text-xl mb-4 flex items-center gap-2 ${textMain}`}>🍚 오늘의 급식</h3>
                  {isOff ? (
                    <div className={`flex items-center justify-center p-8 border-2 ${borderMain} rounded-2xl bg-white ${shadowMd}`}>
                      <p className={`font-black text-center text-lg ${textMain}`}>학교 쉬는 날 🎉</p>
                    </div>
                  ) : noMeals ? (
                    <div className={`flex items-center justify-center p-8 border-2 ${borderMain} rounded-2xl bg-white ${shadowMd}`}>
                      <p className={`font-black text-center text-lg ${textMain}`}>데이터가 없습니다.</p>
                    </div>
                  ) : (
                    Object.entries(meals).map(([mealType, mealList]) => {
                      const voteData = mealDocs[mealType] || {};
                      const likesCount = voteData.likes?.length || 0;
                      const dislikesCount = voteData.dislikes?.length || 0;
                      const iLiked = voteData.likes?.includes(user?.uid);
                      const iDisliked = voteData.dislikes?.includes(user?.uid);

                      return (
                        <div key={mealType} className={`border-2 ${borderMain} rounded-2xl p-6 mb-4 ${shadowMd} bg-white flex flex-col relative overflow-hidden`}>
                          <div className="flex justify-between items-center mb-5 mt-1">
                            {/* 중식/석식 등 태그 화이트모드 및 가시성 완벽 고정 */}
                            <span className={`w-max text-sm font-black px-4 py-1.5 rounded-full border-2 border-black bg-black text-white`}>{mealType}</span>
                          </div>
                          
                          <div className="flex flex-col gap-3 mb-6">
                            {mealList.map((m,i) => <p key={i} className={`font-bold text-lg flex items-center gap-2 ${textMain}`}><span className={`w-2 h-2 rounded-full block bg-black`}></span> {m}</p>)}
                          </div>

                          <div className={`flex gap-3 border-t-2 border-dashed pt-5 ${borderMain}`}>
                             <button onClick={() => handleMealVote(mealType, 'like')} className={`flex-1 py-3 rounded-xl border-2 flex justify-center items-center gap-2 font-black text-sm transition-transform active:translate-y-[2px] ${borderMain} ${iLiked ? 'bg-[#00E066] text-black shadow-sm' : 'bg-gray-50 hover:bg-gray-100 text-gray-500'}`}>
                               <Icon name="thumbs-up" fill={iLiked ? "currentColor" : "none"} size={18} /> 좋아요 {likesCount > 0 && likesCount}
                             </button>
                             <button onClick={() => handleMealVote(mealType, 'dislike')} className={`flex-1 py-3 rounded-xl border-2 flex justify-center items-center gap-2 font-black text-sm transition-transform active:translate-y-[2px] ${borderMain} ${iDisliked ? 'bg-[#FF4444] text-white shadow-sm' : 'bg-gray-50 hover:bg-gray-100 text-gray-500'}`}>
                               <Icon name="thumbs-down" fill={iDisliked ? "currentColor" : "none"} size={18} /> 별로 {dislikesCount > 0 && dislikesCount}
                             </button>
                          </div>
                        </div>
                      )
                    })
                  )}
                </div>

                {/* 오늘 시간표 영역 */}
                <div className={`border-2 ${borderMain} rounded-2xl p-6 md:p-8 mt-2 ${shadowMd} bg-white`}>
                  <h3 className={`font-black text-xl mb-6 flex items-center gap-2 ${textMain}`}>🕒 오늘 시간표</h3>
                  <div className="flex flex-col gap-3">
                    {isOff ? (
                      <div className={`flex items-center justify-center p-6 border-2 ${borderMain} rounded-xl bg-transparent opacity-80`}>
                        <p className={`font-black text-center text-lg ${textMain}`}>학교 쉬는 날 🎉</p>
                      </div>
                    ) : (
                      tt.map((t,i) => (
                        <div key={i} className={`flex justify-between items-center border-2 ${borderMain} rounded-xl p-4 font-bold bg-white`}>
                          <span className={`text-sm px-3 py-1.5 rounded-lg font-black bg-black text-white`}>{i+1}교시</span>
                          <span className={`text-lg ${textMain}`}>{t}</span>
                        </div>
                      ))
                    )}
                  </div>
                </div>
              </>
            )}

            {activeTab === 'ranking' && (
               <div className="flex flex-col gap-5 pb-10">
                  <div className="flex justify-center mb-2">
                     <div className={`bg-gray-100 border-2 ${borderMain} rounded-xl p-1 flex inline-flex`}>
                        <button onClick={() => setRankScope('national')} className={`px-4 py-1.5 text-sm font-black rounded-lg transition-colors ${rankScope === 'national' ? 'bg-white border-2 border-black shadow-sm' : 'text-gray-500 border-2 border-transparent'}`}>전국 랭킹</button>
                        <button onClick={() => setRankScope('regional')} className={`px-4 py-1.5 text-sm font-black rounded-lg transition-colors ${rankScope === 'regional' ? 'bg-white border-2 border-black shadow-sm' : 'text-gray-500 border-2 border-transparent'}`}>지역별 1위</button>
                     </div>
                  </div>

                  {!rankings || rankings.length === 0 ? (
                     <div className="text-center py-16 flex flex-col items-center border-2 border-dashed border-gray-300 rounded-2xl gap-4 bg-white">
                        <Icon name="trophy" size={48} className="text-gray-300"/>
                        <p className="font-bold text-gray-500">아직 오늘의 급식 랭킹이 생성되지 않았습니다.</p>
                        <NeoButton onClick={handleGenerateAIRanking} disabled={isGenerating} className="max-w-[250px] mt-2">
                           <Icon name="sparkles" size={18}/> {isGenerating ? 'AI가 분석 중...' : '오늘의 랭킹 생성하기'}
                        </NeoButton>
                     </div>
                  ) : (
                     <>
                        {rankScope === 'national' && rankings.national && rankings.national.map((r, idx) => (
                             <div key={idx} className={`border-2 ${borderMain} rounded-2xl p-6 flex flex-col ${shadowSm} bg-white relative overflow-hidden transition-transform hover:-translate-y-1`}>
                                 <div className={`absolute top-0 right-0 px-5 py-2 font-black text-base border-b-2 border-l-2 ${borderMain} rounded-bl-2xl ${idx === 0 ? 'bg-[#FFD900]' : idx === 1 ? 'bg-gray-300' : idx === 2 ? 'bg-[#CD7F32] text-white' : 'bg-gray-100'}`}>
                                    {r.rank}위
                                 </div>
                                 <div className="font-black text-xl mb-3 pr-16 text-black flex items-end gap-2">
                                     {r.schoolName}
                                     <span className="text-sm text-gray-500 font-bold mb-0.5">{r.region}</span>
                                 </div>
                                 <div className="text-base font-bold bg-gray-50 border-2 border-black p-4 rounded-xl leading-relaxed text-gray-800 shadow-sm">{r.meal}</div>
                             </div>
                        ))}
                        
                        {rankScope === 'regional' && rankings.regional && rankings.regional.map((r, idx) => (
                             <div key={idx} className={`border-2 ${borderMain} rounded-2xl p-6 flex flex-col ${shadowSm} bg-white relative overflow-hidden transition-transform hover:-translate-y-1`}>
                                 <div className={`absolute top-0 right-0 px-4 py-2 font-black text-sm border-b-2 border-l-2 ${borderMain} rounded-bl-2xl bg-black text-white`}>
                                    📍 {r.region}
                                 </div>
                                 <div className="font-black text-xl mb-3 pr-16 text-black flex items-end gap-2">
                                     {r.schoolName}
                                 </div>
                                 <div className="text-base font-bold bg-gray-50 border-2 border-black p-4 rounded-xl leading-relaxed text-gray-800 shadow-sm">{r.meal}</div>
                             </div>
                        ))}
                     </>
                  )}
               </div>
            )}
          </div>
        </div>
      );
    }

    // =====================================================================
    // [ views/WriteView.jsx ] 게시글 작성 및 수정 뷰 + 게시판 개설 뷰 연동
    // =====================================================================
    function WriteView() {
      const { params, navigate, goBack } = useRouter(); 
      const { showToast } = useUI();
      const { user } = useAuth();
      const { boards } = useUI();
      const { bgApp, bgCard, bgInput, textMain, textMuted, borderMain, shadowMd, shadowSm } = useThemeStyles();
      
      const [wt, setWt] = useState(null); 
      const [title, setTitle] = useState(''); 
      const [content, setContent] = useState(''); 
      const [opts, setOpts] = useState(['','']);
      const [isAnon, setIsAnon] = useState(false);
      const [visibility, setVisibility] = useState('전국');
      const [selectedBoards, setSelectedBoards] = useState([]);

      // 게시판 생성 폼 상태
      const [boardName, setBoardName] = useState('');
      const [boardCategory, setBoardCategory] = useState(BOARD_CATEGORIES[0]);
      const [boardDesc, setBoardDesc] = useState('');
      const [boardImage, setBoardImage] = useState(null);

      const defaultBoard = params.boardId ? decodeURIComponent(params.boardId) : '자유';
      
      const isEditMode = !!params.id;

      useEffect(() => {
        if (isEditMode) {
          getDoc(doc(getPublicCol('posts'), params.id)).then(snap => {
            if(snap.exists()) {
              const d = snap.data();
              if(d.authorId !== user?.uid) { 
                showToast('수정 권한이 없습니다.'); 
                return goBack(); 
              }
              setWt(d.poll ? 'poll' : d.tags?.includes('🙋 질문') ? 'question' : 'general');
              setTitle(d.title || '');
              setContent(d.content || '');
              setIsAnon(d.isAnon || false);
              setVisibility(d.visibility || '전국');
              setSelectedBoards(d.boards || [defaultBoard]);
              if (d.poll) setOpts(d.poll.options.map(o => o.label));
            }
          });
        } else {
          setSelectedBoards([defaultBoard]);
        }
      }, [isEditMode, params.id, user, goBack, showToast, defaultBoard]);

      const toggleBoard = (b) => {
        setSelectedBoards([b]);
      };

      const handleCreateBoard = async () => {
        if (!boardName.trim()) return showToast('게시판 이름을 입력해주세요.');
        if (boards.some(b => b.name === boardName.trim())) return showToast('이미 존재하는 게시판 이름입니다.');
        
        try {
          await createNewBoard({
            name: boardName.trim(),
            category: boardCategory,
            description: boardDesc.trim(),
            image: boardImage,
            creatorId: user.uid
          });
          showToast('게시판이 성공적으로 개설되었습니다!');
          navigate(`board?id=${encodeURIComponent(boardName.trim())}`);
        } catch(e) {
          showToast('게시판 개설 중 오류가 발생했습니다.');
        }
      };

      const handleBoardImageUpload = (e) => {
        const file = e.target.files[0];
        if(file) {
          if(file.size > 1048576) return showToast('1MB 이하의 이미지만 가능합니다.'); 
          const reader = new FileReader();
          reader.onloadend = () => setBoardImage(reader.result);
          reader.readAsDataURL(file);
        }
      };

      if (!wt) {
        return (
          <div className={`flex-1 p-6 md:p-10 flex flex-col h-full gap-5 pt-12 relative animate-view ${bgApp}`}>
            <div className="flex justify-between items-center mb-8">
               <h2 className={`sys-text text-3xl font-black ${textMain}`}>어떤 글을<br/>작성할까요?</h2>
               <button onClick={() => goBack()} className={`p-3 rounded-full border-2 transition-colors ${bgCard} ${borderMain} ${shadowSm} ${textMain}`}>
                 <Icon name="x" size={24}/>
               </button>
            </div>
            <button onClick={() => setWt('general')} className={`p-6 md:p-8 border-2 rounded-2xl font-black text-xl md:text-2xl text-left hover:-translate-y-1 transition-all flex items-center gap-5 bg-white ${borderMain} ${shadowMd} ${textMain}`}>
               <div className={`p-4 rounded-xl border-2 ${borderMain} bg-gray-100`}><Icon name="file-text" size={28}/></div> 일반 글쓰기
            </button>
            <button onClick={() => setWt('question')} className={`p-6 md:p-8 border-2 rounded-2xl font-black text-xl md:text-2xl text-left hover:-translate-y-1 transition-all flex items-center gap-5 bg-white ${borderMain} ${shadowMd} ${textMain}`}>
               <div className={`p-4 rounded-xl border-2 bg-gray-100 ${borderMain}`}><Icon name="help-circle" size={28}/></div> 질문하기
            </button>
            <button onClick={() => setWt('poll')} className={`p-6 md:p-8 border-2 rounded-2xl font-black text-xl md:text-2xl text-left hover:-translate-y-1 transition-all flex items-center gap-5 bg-white ${borderMain} ${shadowMd} ${textMain}`}>
               <div className={`p-4 rounded-xl border-2 bg-gray-100 ${borderMain}`}><Icon name="bar-chart-2" size={28}/></div> 투표 올리기
            </button>
            
            <div className={`border-t-2 ${borderMain} my-4`}></div>
            
            <button onClick={() => setWt('create_board')} className={`p-6 md:p-8 border-2 rounded-2xl font-black text-xl md:text-2xl text-left hover:-translate-y-1 transition-all flex items-center gap-5 bg-white ${borderMain} ${shadowMd} ${textMain}`}>
               <div className={`p-4 rounded-xl border-2 bg-gray-100 text-black ${borderMain}`}><Icon name="layout-dashboard" size={28}/></div> 게시판 개설하기
            </button>
          </div>
        );
      }

      if (wt === 'create_board') {
        return (
          <div className={`flex-1 flex flex-col h-full relative animate-[slideInRight_0.3s_ease-out] ${bgApp}`}>
            <div className={`p-4 md:p-6 border-b-2 flex justify-between items-center z-10 shadow-sm ${bgCard} ${borderMain}`}>
              <button onClick={() => setWt(null)} className={`p-2 -ml-2 rounded-full transition-colors ${textMain} hover:bg-gray-100`}><Icon name="chevron-left" size={28}/></button>
              <span className={`sys-text font-black text-xl ${textMain}`}>게시판 개설</span>
              <button onClick={handleCreateBoard} className={`font-black px-5 py-2 rounded-xl border-2 text-base active:translate-y-[2px] active:translate-x-[2px] active:shadow-none transition-all main-btn ${borderMain} ${shadowSm}`}>개설</button>
            </div>
            <div className="flex-1 p-6 md:p-10 overflow-y-auto hide-scrollbar flex flex-col gap-6 md:gap-8 bg-white">
              <div className="flex flex-col items-center gap-4 mb-4">
                <div className={`w-32 h-32 border-4 rounded-3xl flex items-center justify-center relative overflow-hidden ${borderMain} ${shadowMd} bg-white`}>
                  {boardImage ? (
                    <img src={boardImage} alt="board" className="w-full h-full object-cover" />
                  ) : (
                    <Icon name="image" size={48} className={textMuted}/>
                  )}
                </div>
                <label className={`cursor-pointer px-5 py-2.5 border-2 rounded-xl font-black text-sm shadow-[2px_2px_0px_rgba(0,0,0,1)] active:translate-y-[2px] active:translate-x-[2px] active:shadow-none transition-all ${bgCard} ${textMain} ${borderMain}`}>
                  간판 이미지 등록
                  <input type="file" accept="image/*" className="hidden" onChange={handleBoardImageUpload} />
                </label>
              </div>
              
              <NeoInput label="게시판 이름" value={boardName} onChange={e=>setBoardName(e.target.value)} placeholder="새로운 게시판 이름" />
              
              <div className="flex flex-col gap-2">
                <label className={`font-bold text-sm md:text-base ${textMain}`}>카테고리 선택</label>
                <select value={boardCategory} onChange={e=>setBoardCategory(e.target.value)} className={`w-full p-4 md:p-5 border-2 rounded-xl font-bold text-lg outline-none ${bgInput} ${textMain} ${borderMain}`}>
                  {BOARD_CATEGORIES.map(c => <option key={c} value={c}>{c}</option>)}
                </select>
              </div>

              <NeoInput label="게시판 설명 (선택)" value={boardDesc} onChange={e=>setBoardDesc(e.target.value)} placeholder="게시판의 목적을 간단히 적어주세요." />
            </div>
          </div>
        );
      }

      const handlePostUpload = async () => {
        if (!title.trim() || !content.trim()) return showToast('제목과 내용을 모두 입력해주세요.');
        if (wt === 'poll' && opts.filter(o => o.trim()).length < 2) return showToast('투표 항목을 최소 2개 입력해주세요.');
        if (selectedBoards.length === 0) return showToast('게시판을 최소 1개 선택해주세요.');

        const targetBoard = selectedBoards[0];
        const tags = [wt === 'poll' ? '📊 투표' : wt === 'question' ? '🙋 질문' : '💬 자유', targetBoard];
        const pollData = wt === 'poll' ? { question: title, totalVotes: 0, options: opts.filter(o => o.trim()).map(l => ({ label: l, votes: 0, percent: 0, voters: [] })) } : null;

        let authorName = user?.nickname || '유저';
        if (isAnon) {
          const rand = Math.floor(1000 + Math.random() * 9000);
          authorName = `익명${rand}`;
        }

        const postRegion = user?.school?.address?.split(' ').slice(0, 2).join(' ') || '';

        try {
          if (isEditMode) {
            await updatePost(params.id, { 
              title: title.trim(), content: content.trim(), tags, boards: selectedBoards, boardId: targetBoard, boardName: targetBoard,
              poll: pollData, isAnon, author: authorName, visibility 
            });
            showToast('성공적으로 수정되었습니다!');
            goBack();
          } else {
            await createNewPost({ 
              title: title.trim(), 
              content: content.trim(), 
              tags, 
              boards: selectedBoards,
              boardId: targetBoard,
              boardName: targetBoard,
              poll: pollData, 
              author: authorName,
              authorId: user?.uid,
              schoolId: user?.school?.id || null,
              grade: user?.grade || null,
              classNum: user?.classNum || null,
              region: postRegion,
              visibility,
              isAnon: isAnon,
              profileImage: isAnon ? null : user?.profileImage
            }); 
            showToast('성공적으로 등록되었습니다!'); 
            navigate(`board?id=${encodeURIComponent(targetBoard)}`);
          }
        } catch(e) {
          showToast('처리 중 오류가 발생했습니다.');
        }
      };

      return (
        <div className={`flex-1 flex flex-col h-full relative animate-[slideInRight_0.3s_ease-out] ${bgApp}`}>
          <div className={`p-4 md:p-6 border-b-2 flex justify-between items-center z-10 shadow-sm ${bgCard} ${borderMain}`}>
            <button onClick={() => isEditMode ? goBack() : setWt(null)} className={`p-2 -ml-2 rounded-full transition-colors ${textMain} hover:bg-gray-100`}><Icon name="chevron-left" size={28}/></button>
            <span className={`sys-text font-black text-xl md:text-2xl ${textMain}`}>{isEditMode ? '게시글 수정' : '글쓰기'}</span>
            <button onClick={handlePostUpload} className={`font-black px-5 py-2.5 rounded-xl border-2 text-base active:translate-y-[2px] active:translate-x-[2px] active:shadow-none transition-all main-btn ${borderMain} ${shadowSm}`}>등록</button>
          </div>
          <div className="flex-1 p-5 md:p-8 overflow-y-auto hide-scrollbar flex flex-col gap-5 bg-white">
            <div className="flex items-center justify-between mb-4">
               <span className={`text-xs font-black px-3 py-1.5 rounded-md border ${borderMain} bg-gray-200 text-black`}>{wt === 'poll' ? '📊 투표' : wt === 'question' ? '🙋 질문' : '💬 자유'}</span>
               <div className="flex items-center gap-5">
                 <select value={visibility} onChange={e => setVisibility(e.target.value)} className={`p-2 text-sm font-black border-2 rounded-md outline-none cursor-pointer bg-white ${borderMain} ${textMain}`}>
                   <option value="전국">🌐 전국</option>
                   <option value="지역구">🗺️ 지역구</option>
                   <option value="우리 학교">🏫 우리 학교</option>
                   <option value="우리 반">🚪 우리 반</option>
                 </select>
                 <div className="flex items-center gap-2">
                   <input type="checkbox" id="anon-post-check" checked={isAnon} onChange={e => setIsAnon(e.target.checked)} className={`accent-black w-5 h-5 border-2 cursor-pointer ${borderMain}`}/>
                   <label htmlFor="anon-post-check" className={`text-sm font-black cursor-pointer ${textMuted}`}>익명</label>
                 </div>
               </div>
            </div>

            <h3 className={`font-bold text-base ${textMuted}`}>게시판 선택</h3>
            <div className="flex flex-wrap gap-3 mb-4">
              {boards.map(b => (
                <button 
                  key={b.id}
                  onClick={() => toggleBoard(b.name)}
                  className={`px-4 py-2 border-2 rounded-lg font-bold text-sm transition-colors ${borderMain} ${selectedBoards.includes(b.name) ? 'bg-black text-white' : `bg-transparent ${textMuted}`}`}
                >{b.name}</button>
              ))}
            </div>

            <input 
              placeholder="제목을 입력하세요" 
              value={title} 
              onChange={e => setTitle(e.target.value)}

              className={`w-full h-64 font-bold outline-none resize-none bg-transparent text-lg placeholder:opacity-50 mt-4 ${textMain}`}
            ></textarea>
            
            {wt === 'poll' && (
              <div className={`border-2 rounded-2xl p-6 mt-6 ${bgCard} ${borderMain} ${shadowSm}`}>
                <h3 className={`font-black mb-5 flex items-center gap-2 text-base ${textMain}`}><Icon name="bar-chart-2" size={20}/> 투표 항목 구성</h3>

                      <input 
                        value={o} 
                        onChange={e => { const n=[...opts]; n[i]=e.target.value; setOpts(n); }} 
                        placeholder="항목 내용을 입력하세요"
                        className={`flex-1 p-4 border-2 rounded-xl outline-none font-bold text-base transition-shadow focus:shadow-[2px_2px_0px_rgba(0,0,0,1)] ${bgInput} ${borderMain} ${textMain}`}
                      />
                      <button onClick={() => { if(opts.length > 2) setOpts(opts.filter((_,x) => x !== i)); }} className={`p-4 border-2 rounded-xl transition-colors ${borderMain} bg-gray-50 hover:bg-red-50 text-gray-400 hover:text-red-500`}>
                        <Icon name="trash-2" size={20} />

                <button onClick={() => opts.length < 8 && setOpts([...opts,''])} className={`w-full p-4 border-2 border-dashed font-bold rounded-xl mt-6 transition-colors flex items-center justify-center gap-2 ${borderMain} ${textMuted} hover:bg-gray-50 hover:text-black hover:border-black`}>

    // [ views/SettingsView.jsx ] 설정 및 프로필 변경 뷰 (목록형 구조 개편)
    // =====================================================================
    function SettingsView() {
      const { user } = useAuth(); 
      const { navigate } = useRouter(); 
      const { showModal, showToast } = useUI();
      const { settings, updateSetting } = useSettings();
      const { bgApp, bgCard, textMain, textMuted, borderMain, shadowSm, borderLight } = useThemeStyles();
      
      const [settingsSubTab, setSettingsSubTab] = useState('menu'); // 'menu' | 'edit-profile' | 'ui-setting' | 'school-friends' | 'my-likes' | 'notification-setting' | 'device-setting'
      
      const [editForm, setEditForm] = useState({
        id: user?.id || '', 
        nickname: user?.nickname || '', 
        bio: user?.bio || '',
        grade: user?.grade || 1, 
        classNum: user?.classNum || 1
      });

      // 마스킹된 우리 학교 친구 목록 상태
      const [schoolFriends, setSchoolFriends] = useState([]);
      // 내가 좋아요 누른 게시글 목록
      const { posts } = usePosts();
      const likedPosts = useMemo(() => {
         return posts.filter(p => p.likedBy?.includes(user?.uid));
      }, [posts, user]);

      useEffect(() => {
        if(user) setEditForm({ 
          id: user.id || '', 
          nickname: user.nickname || '', 
          bio: user.bio || '',
          grade: user.grade || 1, 
          classNum: user.classNum || 1
        });
      }, [user]);

      // 우리 학교 친구 목록 불러오기 (마스킹 처리 반영)
      useEffect(() => {
         if (settingsSubTab === 'school-friends' && user?.school?.id) {
            getDocs(query(getPublicCol('users'), where('school.id', '==', user.school.id))).then(snap => {
               const list = snap.docs.map(d => {
                  const data = d.data();
                  // 이름 마스킹 처리 (김*현)
                  let maskedName = data.nickname || "친구";
                  if (maskedName.length > 2) {
                     maskedName = maskedName[0] + '*'.repeat(maskedName.length - 2) + maskedName[maskedName.length - 1];
                  } else if (maskedName.length === 2) {
                     maskedName = maskedName[0] + '*';
                  }
                  return { uid: d.id, nickname: maskedName, id: data.id };
               });
               setSchoolFriends(list);
            });
         }
      }, [settingsSubTab, user]);

      const handleSaveProfile = async () => {
        if (!editForm.id || !editForm.nickname) return showToast('아이디와 닉네임을 입력해주세요.');
        
        const now = Date.now();
        const ONE_MONTH = 30 * 24 * 60 * 60 * 1000;
        const SIX_MONTHS = 180 * 24 * 60 * 60 * 1000;
        let updates = { ...editForm };

        if (editForm.id !== user.id) {
          if (user.lastIdChange && (now - user.lastIdChange < ONE_MONTH)) {
            return showToast('아이디는 1개월에 한 번만 변경할 수 있습니다.');
          }
          updates.lastIdChange = now;
        }

        if (editForm.grade !== user.grade || editForm.classNum !== user.classNum) {
          const history = user.schoolChangeHistory || [];
          const recentChanges = history.filter(ts => now - ts < SIX_MONTHS);
          if (recentChanges.length >= 2) {
            return showToast('학년/반 정보는 6개월에 2번만 변경할 수 있습니다.');
          }
          updates.schoolChangeHistory = [...recentChanges, now];
        }

        await updateUserProfile(user.uid, updates);
        showToast('내 정보가 성공적으로 수정되었습니다.');
        setSettingsSubTab('menu');
      };

      const handleSchoolChange = () => {
        const now = Date.now();
        const history = user.schoolChangeHistory || [];
        const recentChanges = history.filter(ts => now - ts < 180 * 24 * 60 * 60 * 1000);
        if (recentChanges.length >= 2) {
            return showToast('학교 정보는 6개월에 2번만 변경할 수 있습니다.');
        }
        navigate('school');
      };

      // 알림 종류별 ON/OFF 기본 상태 설정
      const [notifSettings, setNotifSettings] = useState(() => {
         const saved = localStorage.getItem('ocl_notif_settings');
         return saved ? JSON.parse(saved) : { comments: true, likes: true, reaction: true, recommendation: true, meal: true };
      });

      const handleToggleNotif = (key) => {
         const updated = { ...notifSettings, [key]: !notifSettings[key] };
         setNotifSettings(updated);
         localStorage.setItem('ocl_notif_settings', JSON.stringify(updated));
         showToast('알림 설정이 변경되었습니다.');

         // Native WebView Bridge Push Notification sync
         if (isAppEnvironment()) {
            window.ReactNativeWebView.postMessage(JSON.stringify({
               type: 'SYNC_PUSH_PREFERENCE',
               preference: updated
            }));
         }
      };

      // 목록형 설정화면 메인 구조
      if (settingsSubTab === 'menu') {
         return (
           <div className={`flex-1 flex flex-col h-full animate-[slideInRight_0.3s_ease-out] ${bgApp}`}>
             <div className={`p-4 md:p-6 border-b-2 flex items-center justify-between z-10 shadow-sm ${bgCard} ${borderMain}`}>
               <div className="flex items-center gap-3">
                 <button onClick={() => navigate('my')} className="p-2 -ml-2 rounded-full transition-colors text-black hover:bg-gray-100 w-10 h-10 flex shrink-0 items-center justify-center">
                   <Icon name="chevron-left" size={28} strokeWidth={3}/>
                 </button>
                 <span className={`font-black text-xl md:text-2xl ${textMain}`}>설정</span>
               </div>
             </div>

             <div className="flex-1 px-6 md:px-8 py-6 flex flex-col gap-2 overflow-y-auto hide-scrollbar bg-white">
                <button onClick={() => setSettingsSubTab('edit-profile')} className="w-full text-left p-4 border-2 border-black rounded-xl font-black hover:bg-gray-50 flex justify-between items-center">
                   <span>정보 변경</span>
                   <Icon name="chevron-right" size={18} />
                </button>

                {/* 앱 환경에서만 노출 설정 활성화 */}
                {isAppEnvironment() && (
                  <button onClick={() => setSettingsSubTab('ui-setting')} className="w-full text-left p-4 border-2 border-black rounded-xl font-black hover:bg-gray-50 flex justify-between items-center">
                     <span>UI 설정</span>
                     <Icon name="chevron-right" size={18} />
                  </button>
                )}

                {user?.school && (
                   <button onClick={() => setSettingsSubTab('school-friends')} className="w-full text-left p-4 border-2 border-black rounded-xl font-black hover:bg-gray-50 flex justify-between items-center">
                      <span>우리 학교 친구 보기</span>
                      <Icon name="chevron-right" size={18} />
                   </button>
                )}

                <a href="https://vocal-kitten-d0c02e.netlify.app" target="_blank" rel="noopener noreferrer" className="w-full text-left p-4 border-2 border-black rounded-xl font-black hover:bg-gray-50 flex justify-between items-center">
                   <span>개인정보 처리방침</span>
                   <Icon name="chevron-right" size={18} />
                </a>

                <button onClick={() => setSettingsSubTab('my-likes')} className="w-full text-left p-4 border-2 border-black rounded-xl font-black hover:bg-gray-50 flex justify-between items-center">
                   <span>나의 좋아요</span>
                   <Icon name="chevron-right" size={18} />
                </button>

                <button onClick={() => setSettingsSubTab('notification-setting')} className="w-full text-left p-4 border-2 border-black rounded-xl font-black hover:bg-gray-50 flex justify-between items-center">
                   <span>알림 설정</span>
                   <Icon name="chevron-right" size={18} />
                </button>

                {isAppEnvironment() && (
                   <button onClick={() => setSettingsSubTab('device-setting')} className="w-full text-left p-4 border-2 border-black rounded-xl font-black hover:bg-gray-50 flex justify-between items-center">
                      <span>로그인 기기 설정</span>
                      <Icon name="chevron-right" size={18} />
                   </button>
                )}

                <button onClick={() => showModal('정말 로그아웃 하시겠습니까?', 'error', async () => { await signOut(auth); window.location.reload(); }, '로그아웃')} className="w-full text-left p-4 border-2 border-[#FF4444] rounded-xl font-black hover:bg-red-50 flex justify-between items-center text-[#FF4444] mt-4">
                   <span>로그아웃</span>
                   <Icon name="logout" size={18} />
                </button>
             </div>
           </div>
         );
      }

      // 서브 탭 1: 정보 변경
      if (settingsSubTab === 'edit-profile') {
         return (
           <div className={`flex-1 flex flex-col h-full animate-[slideInRight_0.3s_ease-out] ${bgApp}`}>
             <div className={`p-4 md:p-6 border-b-2 flex items-center justify-between z-10 shadow-sm ${bgCard} ${borderMain}`}>
               <div className="flex items-center gap-3">
                 <button onClick={() => setSettingsSubTab('menu')} className="p-2 -ml-2 rounded-full transition-colors text-black hover:bg-gray-100 w-10 h-10 flex shrink-0 items-center justify-center">
                   <Icon name="chevron-left" size={28} strokeWidth={3}/>
                 </button>
                 <span className={`font-black text-xl ${textMain}`}>정보 변경</span>
               </div>
               <button onClick={handleSaveProfile} className={`font-black px-5 py-2 rounded-xl border-2 text-base main-btn ${borderMain} ${shadowSm}`}>저장</button>
             </div>
             <div className="flex-1 px-6 py-6 bg-white overflow-y-auto hide-scrollbar flex flex-col gap-6">
               <NeoInput label="아이디 (1개월 1회 변경 가능)" value={editForm.id} onChange={e => setEditForm({...editForm, id: e.target.value})} />
               <NeoInput label="닉네임" value={editForm.nickname} onChange={e => setEditForm({...editForm, nickname: e.target.value})} />
               <NeoInput label="소개글" value={editForm.bio} onChange={e => setEditForm({...editForm, bio: e.target.value})} placeholder="나를 짧게 소개해보세요!" />

               <div className="flex flex-col gap-5 mt-4">
                  <h3 className={`font-black text-base border-b-2 pb-3 ${textMuted} ${borderMain}`}>학교 정보</h3>
                  <div className={`flex justify-between items-center p-5 border-2 rounded-xl shadow-sm ${bgCard} ${borderMain}`}>
                     <span className={`font-bold text-lg ${textMain}`}>{user?.school?.name || '학교 정보 없음'}</span>
                     <button onClick={handleSchoolChange} className={`text-sm font-black px-4 py-2 rounded-lg border-2 main-btn ${borderMain}`}>학교 변경</button>
                  </div>
                  <div className="flex gap-5">
                     <NeoInput label="학년" type="number" value={editForm.grade} onChange={e => setEditForm({...editForm, grade: Number(e.target.value)})} />
                     <NeoInput label="반" type="number" value={editForm.classNum} onChange={e => setEditForm({...editForm, classNum: Number(e.target.value)})} />
                  </div>
               </div>
             </div>
           </div>
         );
      }

      // 서브 탭 2: UI 설정 (앱에서만 활성)
      if (settingsSubTab === 'ui-setting') {
         return (
           <div className={`flex-1 flex flex-col h-full animate-[slideInRight_0.3s_ease-out] ${bgApp}`}>
             <div className={`p-4 md:p-6 border-b-2 flex items-center justify-between z-10 shadow-sm ${bgCard} ${borderMain}`}>
               <div className="flex items-center gap-3">
                 <button onClick={() => setSettingsSubTab('menu')} className="p-2 -ml-2 rounded-full transition-colors text-black hover:bg-gray-100 w-10 h-10 flex shrink-0 items-center justify-center">
                   <Icon name="chevron-left" size={28} strokeWidth={3}/>
                 </button>
                 <span className={`font-black text-xl ${textMain}`}>UI 설정</span>
               </div>
             </div>
             <div className="flex-1 px-6 py-6 bg-white overflow-y-auto hide-scrollbar flex flex-col gap-6">
                <div className="flex flex-col gap-2">
                  <span className="font-bold text-sm">테마 모드</span>
                  <div className="flex gap-2">
                    {['light', 'dark', 'device'].map(t => (
                      <button key={t} onClick={() => { updateSetting('theme', t); showToast('설정이 저장되었습니다.'); }} className={`flex-1 py-3 border-2 border-black rounded-xl font-bold ${settings.theme === t ? 'bg-[#FFD900]' : 'bg-white'}`}>
                        {t === 'light' ? '라이트' : t === 'dark' ? '다크' : '기기설정'}
                      </button>
                    ))}
                  </div>
                </div>

                <div className="flex flex-col gap-2">
                  <span className="font-bold text-sm">글씨 크기</span>
                  <div className="flex gap-2">
                    {['small', 'medium', 'large'].map(f => (
                      <button key={f} onClick={() => { updateSetting('fontSize', f); showToast('설정이 저장되었습니다.'); }} className={`flex-1 py-3 border-2 border-black rounded-xl font-bold ${settings.fontSize === f ? 'bg-[#FFD900]' : 'bg-white'}`}>
                        {f === 'small' ? '작게' : f === 'medium' ? '표준' : '크게'}
                      </button>
                    ))}
                  </div>
                </div>

                <div className="flex flex-col gap-2">
                  <span className="font-bold text-sm">애니메이션 효과</span>
                  <button onClick={() => { updateSetting('animation', settings.animation === 'none' ? 'default' : 'none'); showToast('설정이 저장되었습니다.'); }} className={`w-full py-3 border-2 border-black rounded-xl font-bold ${settings.animation !== 'none' ? 'bg-[#FFD900]' : 'bg-white'}`}>
                    {settings.animation !== 'none' ? '애니메이션 사용 중' : '애니메이션 효과 없음'}
                  </button>
                </div>
             </div>
           </div>
         );
      }

      // 서브 탭 3: 우리 학교 친구 보기 (이름 일부 마스킹 처리 김*현 반영)
      if (settingsSubTab === 'school-friends') {
         return (
           <div className={`flex-1 flex flex-col h-full animate-[slideInRight_0.3s_ease-out] ${bgApp}`}>
             <div className={`p-4 md:p-6 border-b-2 flex items-center justify-between z-10 shadow-sm ${bgCard} ${borderMain}`}>
               <div className="flex items-center gap-3">
                 <button onClick={() => setSettingsSubTab('menu')} className="p-2 -ml-2 rounded-full transition-colors text-black hover:bg-gray-100 w-10 h-10 flex shrink-0 items-center justify-center">
                   <Icon name="chevron-left" size={28} strokeWidth={3}/>
                 </button>
                 <span className={`font-black text-xl ${textMain}`}>우리 학교 친구 보기</span>
               </div>
             </div>
             <div className="flex-1 px-6 py-6 bg-white overflow-y-auto hide-scrollbar flex flex-col gap-4">
                <p className="text-xs text-gray-500 font-bold">개인정보 보호를 위해 친구들의 실명 및 닉네임 일부가 마스킹 처리되어 노출됩니다.</p>
                {schoolFriends.length === 0 ? (
                   <p className="text-center py-10 font-bold text-gray-400">아직 가입한 친구가 없습니다.</p>
                ) : (
                   schoolFriends.map(f => (
                      <div key={f.uid} className="flex justify-between items-center p-4 border-2 border-black rounded-xl shadow-sm bg-white">
                         <div className="flex items-center gap-3">
                            <div className="w-8 h-8 rounded-full bg-gray-200 flex items-center justify-center border border-black"><Icon name="user" size={16} /></div>
                            <span className="font-bold text-base">{f.nickname}</span>
                         </div>
                         <span className="text-xs text-gray-400">@{f.id}</span>
                      </div>
                   ))
                )}
             </div>
           </div>
         );
      }

      // 서브 탭 4: 나의 좋아요
      if (settingsSubTab === 'my-likes') {
         return (
           <div className={`flex-1 flex flex-col h-full animate-[slideInRight_0.3s_ease-out] ${bgApp}`}>
             <div className={`p-4 md:p-6 border-b-2 flex items-center justify-between z-10 shadow-sm ${bgCard} ${borderMain}`}>
               <div className="flex items-center gap-3">
                 <button onClick={() => setSettingsSubTab('menu')} className="p-2 -ml-2 rounded-full transition-colors text-black hover:bg-gray-100 w-10 h-10 flex shrink-0 items-center justify-center">
                   <Icon name="chevron-left" size={28} strokeWidth={3}/>
                 </button>
                 <span className={`font-black text-xl ${textMain}`}>나의 좋아요</span>
               </div>
             </div>
             <div className="flex-1 px-6 py-6 bg-white overflow-y-auto hide-scrollbar flex flex-col gap-4">
                {likedPosts.length === 0 ? (
                   <div className="text-center py-16 text-gray-400 font-bold">좋아요를 표시한 게시글이 없습니다.</div>
                ) : (
                   likedPosts.map(p => (
                      <div key={p.id} onClick={() => navigate(`post?id=${p.id}`)} className="p-4 border-2 border-black rounded-xl bg-white cursor-pointer hover:bg-gray-50 flex flex-col gap-1">
                         <span className="text-xs text-gray-400 font-bold">{formatTime(p.createdAt)}</span>
                         <h4 className="font-black text-lg text-black truncate">{p.title}</h4>
                      </div>
                   ))
                )}
             </div>
           </div>
         );
      }

      // 서브 탭 5: 알림 설정 (FCM 기반 개별 ON/OFF 대응 및 권한 요청 대응)
      if (settingsSubTab === 'notification-setting') {
         const handleRequestPermission = () => {
            if (isAppEnvironment()) {
               window.ReactNativeWebView.postMessage(JSON.stringify({
                  type: 'REQUEST_NOTIFICATION_PERMISSION'
               }));
            } else {
               showToast('웹 환경에서는 모의 푸시알림 권한이 허용되었습니다.');
            }
         };

         return (
           <div className={`flex-1 flex flex-col h-full animate-[slideInRight_0.3s_ease-out] ${bgApp}`}>
             <div className={`p-4 md:p-6 border-b-2 flex items-center justify-between z-10 shadow-sm ${bgCard} ${borderMain}`}>
               <div className="flex items-center gap-3">
                 <button onClick={() => setSettingsSubTab('menu')} className="p-2 -ml-2 rounded-full transition-colors text-black hover:bg-gray-100 w-10 h-10 flex shrink-0 items-center justify-center">
                   <Icon name="chevron-left" size={28} strokeWidth={3}/>
                 </button>
                 <span className={`font-black text-xl ${textMain}`}>알림 설정</span>
               </div>
             </div>
             <div className="flex-1 px-6 py-6 bg-white overflow-y-auto hide-scrollbar flex flex-col gap-6">
                <div className="p-4 border-2 border-dashed border-black rounded-xl bg-gray-50 flex items-center justify-between">
                   <div className="flex flex-col">
                      <span className="font-black text-base text-black">알림 권한 허용</span>
                      <span className="text-xs text-gray-500 font-bold">FCM 알림 수신을 위한 기기 권한을 요청합니다.</span>
                   </div>
                   <button onClick={handleRequestPermission} className="px-4 py-2 border-2 border-black bg-[#FFD900] font-black text-xs rounded-lg shadow-sm">권한 요청</button>
                </div>

                <div className="flex flex-col gap-4 mt-2">
                   {[
                      { key: 'comments', label: '댓글 알림', desc: '내 글에 댓글이 작성되었을 때 알림' },
                      { key: 'likes', label: '좋아요 알림', desc: '내 글에 좋아요 반응이 있을 때 알림' },
                      { key: 'reaction', label: '내 글 반응', desc: '투표 결과나 참여 반응 알림' },
                      { key: 'recommendation', label: '추천 게시물', desc: '인기 폭발 추천 게시물 소식 알림' },
                      { key: 'meal', label: '급식 랭킹 알림', desc: '급식 랭킹 생성 시 인기 알림' }
                   ].map(item => (
                      <div key={item.key} className="flex justify-between items-center p-4 border-2 border-black rounded-xl bg-white">
                         <div className="flex flex-col">
                            <span className="font-black text-base text-black">{item.label}</span>
                            <span className="text-xs text-gray-400 font-bold">{item.desc}</span>
                         </div>
                         <button onClick={() => handleToggleNotif(item.key)} className={`w-12 h-6 border-2 border-black rounded-full p-0.5 transition-colors ${notifSettings[item.key] ? 'bg-[#00E066]' : 'bg-gray-200'}`}>
                            <div className={`w-4 h-4 bg-white border border-black rounded-full transition-transform ${notifSettings[item.key] ? 'translate-x-6' : ''}`} />
                         </button>
                      </div>
                   ))}
                </div>
             </div>
           </div>
         );
      }

      // 서브 탭 6: 로그인 기기 설정 및 앱 잠금 (앱 전용)
      if (settingsSubTab === 'device-setting') {
         const [appLockEnabled, setAppLockEnabled] = useState(false);

         const handleToggleAppLock = () => {
            setAppLockEnabled(!appLockEnabled);
            if (isAppEnvironment()) {
               window.ReactNativeWebView.postMessage(JSON.stringify({
                  type: 'SET_APP_LOCK',
                  enabled: !appLockEnabled
               }));
            }
            showToast(!appLockEnabled ? '생체 인식 앱 잠금이 활성화되었습니다.' : '앱 잠금이 비활성화되었습니다.');
         };

         return (
           <div className={`flex-1 flex flex-col h-full animate-[slideInRight_0.3s_ease-out] ${bgApp}`}>
             <div className={`p-4 md:p-6 border-b-2 flex items-center justify-between z-10 shadow-sm ${bgCard} ${borderMain}`}>
               <div className="flex items-center gap-3">
                 <button onClick={() => setSettingsSubTab('menu')} className="p-2 -ml-2 rounded-full transition-colors text-black hover:bg-gray-100 w-10 h-10 flex shrink-0 items-center justify-center">
                   <Icon name="chevron-left" size={28} strokeWidth={3}/>
                 </button>
                 <span className={`font-black text-xl ${textMain}`}>로그인 기기 설정</span>
               </div>
             </div>
             <div className="flex-1 px-6 py-6 bg-white overflow-y-auto hide-scrollbar flex flex-col gap-6">
                <div className="flex flex-col gap-3">
                   <h4 className="font-black text-sm text-gray-500">연결된 기기 목록</h4>
                   <div className="p-4 border-2 border-black rounded-xl bg-white flex justify-between items-center">
                      <div className="flex flex-col">
                         <span className="font-black text-base text-black">현재 기기 (iPhone / Android)</span>
                         <span className="text-xs text-gray-400 font-bold">마지막 접속: 방금 전</span>
                      </div>
                      <span className="text-xs font-bold text-[#00E066] bg-green-50 px-2 py-1 rounded border border-[#00E066]">사용중</span>
                   </div>
                </div>

                <div className="flex flex-col gap-3 mt-2">
                   <h4 className="font-black text-sm text-gray-500">앱 잠금 설정</h4>
                   <div className="p-4 border-2 border-black rounded-xl bg-white flex justify-between items-center">
                      <div className="flex flex-col">
                         <span className="font-black text-base text-black">생체인식 앱 잠금</span>
                         <span className="text-xs text-gray-400 font-bold">Face ID 또는 지문인식으로 앱을 잠급니다.</span>
                      </div>
                      <button onClick={handleToggleAppLock} className={`w-12 h-6 border-2 border-black rounded-full p-0.5 transition-colors ${appLockEnabled ? 'bg-[#00E066]' : 'bg-gray-200'}`}>
                         <div className={`w-4 h-4 bg-white border border-black rounded-full transition-transform ${appLockEnabled ? 'translate-x-6' : ''}`} />
                      </button>
                   </div>
                </div>

                <div className="flex flex-col gap-3 mt-4">
                   <button onClick={() => showToast('다른 모든 기기에서 로그아웃 처리가 완료되었습니다.')} className="w-full py-4 border-2 border-black rounded-xl font-black bg-gray-100 hover:bg-gray-200 text-center"> 다른 모든 기기 로그아웃 </button>
                </div>
             </div>
           </div>
         );
      }

      return null;
    }

    // =====================================================================
    // [ App.jsx ] 애플리케이션 진입점 및 전역 상태 프로바이더
    // =====================================================================
    function App() {
      const [route, setRoute] = useState({ path: 'splash', params: {} });
      const [currentUser, setCurrentUser] = useState(null);
      const [globalBoards, setGlobalBoards] = useState([]);
      
      const [uiState, setUiState] = useState({ 
        toast: { msg: '', exiting: false }, 
        modal: { open: false }, 
        shareState: { open: false }, 
        profileUser: null,
        report: { open: false, postId: null, reportedId: null, content: null },
        profileImgMenu: false
      });

      useEffect(() => {
        if (!currentUser) return; 
        const unsub = onSnapshot(getPublicCol('boards'), (snap) => {
          setGlobalBoards(snap.docs.map(d => ({ id: d.id, ...d.data() })));
        }, (error) => console.log("Boards fetch error:", error));
        return () => unsub();
      }, [currentUser]);

      // 자동 로그인 상태 보존 로직 추가
      useEffect(() => {
        let unsubUser = null;
        const unsubscribe = onAuthStateChanged(auth, (user) => {
          if (user) {
            unsubUser = onSnapshot(doc(getPublicCol('users'), user.uid), (docSnap) => {
              if (docSnap.exists()) {
                setCurrentUser({ uid: user.uid, ...docSnap.data() });
              } else {
                setCurrentUser({ uid: user.uid });
              }
            }, (e) => {
              console.error("Auth profile fetch error:", e);
              setCurrentUser({ uid: user.uid });
            });
          } else {
            setCurrentUser(null);
            if (unsubUser) unsubUser();
          }
        });
        return () => {
          unsubscribe();
          if (unsubUser) unsubUser();
        };
      }, []);

      const navigate = (path) => {
        if (path.includes('?')) { 
          const [p, q] = path.split('?'); 
          const params = Object.fromEntries(new URLSearchParams(q)); 
          setRoute({ path: p, params }); 
        } else {
          setRoute({ path, params: {} });
        }
      };

      const login = async (method, formData) => {
        try {
          let r;
          if (method === 'google') r = await signInWithPopup(auth, new GoogleAuthProvider());
          else if (method === 'apple') r = await signInWithPopup(auth, new OAuthProvider('apple.com'));
          else if (method === 'email' && formData) {
            const em = `${formData.id}@ocl-lounge.app`;
            try { r = await signInWithEmailAndPassword(auth, em, formData.pw); }
            catch (e) { r = await createUserWithEmailAndPassword(auth, em, formData.pw); }
          }
          if (!r) return null;
          const ud = await getDoc(doc(getPublicCol('users'), r.user.uid));
          const finalUser = ud.exists() ? { uid: r.user.uid, ...ud.data() } : { uid: r.user.uid };
          setCurrentUser(finalUser);
          return finalUser;
        } catch (e) {
          setUiState(p => ({ ...p, modal: { open: true, msg: "인증에 실패했습니다. 입력 정보를 확인해 주세요.", type: 'error' } }));
          return null;
        }
      };

      const PageComponent = {
        splash: OnboardingView, 'auth-method': OnboardingView, 'email-form': OnboardingView, 'email-signup': OnboardingView, 'nickname': OnboardingView, 'school': OnboardingView, 'grade': OnboardingView, 'complete': OnboardingView,
        'board-school': OnboardingView, 'board-national': OnboardingView, 'theme-select': OnboardingView,
        dashboard: HomeView, board: BoardView, search: SearchView, post: DetailView, write: WriteView, 'school-info': SchoolView, my: ProfileView, settings: SettingsView
      }[route.path] || HomeView;

      const showNavRoutes = ['dashboard', 'school-info', 'search', 'my', 'board', 'post'];
      const hasBottomNav = showNavRoutes.includes(route.path);
      const isAuthScreen = ['splash', 'auth-method', 'email-form', 'email-signup', 'nickname', 'school', 'grade', 'board-school', 'board-national', 'theme-select'].includes(route.path);

      return (
        <AuthContext.Provider value={{ user: currentUser, login }}>
          <RouterContext.Provider value={{ currentRoute: route.path, params: route.params, navigate, goBack: () => navigate('dashboard') }}>
            <SettingsProvider>
              <UIContext.Provider value={{ 
                boards: globalBoards,
                showToast: (m) => {
                  setUiState(p => ({...p, toast: { msg: m, exiting: false }}));
                  setTimeout(() => {
                    setUiState(p => ({...p, toast: { msg: m, exiting: true }}));
                    setTimeout(() => setUiState(p => p.toast.msg === m ? {...p, toast: { msg: '', exiting: false }} : p), 300);
                  }, 2500);
                }, 
                showModal: (m,t,c) => setUiState(p => ({...p, modal: {open:true, msg:m, type:t, onConfirm:c}})),
                openProfile: async (uid) => { 
                  const d = await getUserProfile(uid); 
                  setUiState(p => ({...p, profileUser: d})) 
                },
                closeProfile: () => setUiState(p => ({...p, profileUser: null})),
                openShare: (title, url) => setUiState(p => ({...p, shareState: { open: true, title, url }})),
                closeShare: () => setUiState(p => ({...p, shareState: { open: false }})),
                openReport: (postId, reportedId, content) => setUiState(p => ({...p, report: { open: true, postId, reportedId, content }})),
                closeReport: () => setUiState(p => ({...p, report: { open: false }})),
                openProfileImgMenu: () => setUiState(p => ({...p, profileImgMenu: true})),
                closeProfileImgMenu: () => setUiState(p => ({...p, profileImgMenu: false})),
                profileUser: uiState.profileUser,
                shareState: uiState.shareState,
                reportState: uiState.report,
                profileImgMenu: uiState.profileImgMenu
              }}>
                
                {/* 반응형 래퍼 (Web / Mobile 적응형 - 항상 화이트 모드 및 배경 완벽 통일) */}
                <div className={`w-full min-h-[100dvh] flex flex-col items-start justify-start font-sans bg-app-white transition-colors`}>
                  <style>{customStyles}</style>
                  
                  {/* 토스트 및 모달 팝업들 */}
                  {uiState.toast.msg && (
                    <div className={`fixed top-12 md:top-20 inset-x-0 mx-auto z-[9999] pointer-events-none flex justify-center w-max max-w-[90%] ${uiState.toast.exiting ? 'toast-exit' : 'toast-enter'}`}>
                      <div className="bg-black text-white px-6 py-3 rounded-full text-sm font-black shadow-xl whitespace-nowrap border-2 border-black pointer-events-auto text-center">
                        {uiState.toast.msg}
                      </div>
                    </div>
                  )}

                  <NeoModal {...uiState.modal} onClose={() => setUiState(p => ({...p, modal: {...p.modal, open: false}}))} />
                  <ProfileModal />
                  <ShareModal />
                  <ProfileImgModal />
                  <ReportModal />
                  
                  {/* === 반응형 레이아웃 구조 === */}
                  <div className="flex w-full h-[100dvh] overflow-hidden max-w-[1000px] mx-auto justify-start items-start">
                     
                     {/* 1. 좌측 데스크톱 사이드바 */}
                     <DesktopNav />

                     {/* 2. 메인 콘텐츠 영역 (사이드바 삭제로 너비 확장) */}
                     <div className={`flex-1 flex flex-col h-[100dvh] relative w-full items-start justify-start ${isAuthScreen ? 'max-w-[420px] md:max-w-[500px]' : 'max-w-[100vw] md:max-w-[650px] border-x-0 md:border-x-2 border-black'}`}>
                       <div className="absolute inset-0 flex flex-col bg-white">
                         
                         <div className={`flex-1 relative w-full h-full overflow-hidden ${hasBottomNav ? 'pb-[64px] md:pb-0' : ''}`}>
                            <PageComponent />
                         </div>
                         
                         {/* 3. 모바일 하단 네비게이션 */}
                         {hasBottomNav && <BottomNav />}

                       </div>
                     </div>

                  </div>
                  {/* === 구조 끝 === */}

                </div>

              </UIContext.Provider>
            </SettingsProvider>
          </RouterContext.Provider>
        </AuthContext.Provider>
      );
    }

    // 마운트
    const root = createRoot(document.getElementById('root'));
    root.render(<App />);
  </script>
